# Demo #9 — Skills / Progressive Knowledge Loading — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Prove Colmena's `load_skill` holds context tokens ~flat and accuracy high as a knowledge corpus grows to 50 nested packs, while naive prompt-stuffing explodes in tokens and degrades, and idiomatic RAG ties on tokens but mis-retrieves.

**Architecture:** A shared `bench_common.scenario_skills` module generates a deterministic, info-dense corpus of nested-reference knowledge packs from single-source-of-truth Python objects (the same objects drive the ground-truth reference functions, so markdown and answer key cannot drift). A Colmena DAG exposes the corpus via `skills_path`; competitor handlers implement a `naive` (prompt-stuff) arm (all 5 frameworks) and an idiomatic `rag` arm (LlamaIndex + LangChain). A serial driver sweeps frameworks × arms × pack-counts {5,20,50} × seeds under one managed LiteLLM proxy, measuring tokens from proxy spans (Colmena by session-file line-delta) and accuracy by numeric reference functions.

**Tech Stack:** Python 3.11, pandas, Colmena PyO3 binding (`colmena.run_dag`), LiteLLM proxy, LlamaIndex `VectorStoreIndex`, LangChain `VectorStoreRetriever`, matplotlib, pytest.

**Spec:** [docs/superpowers/specs/2026-06-21-demo09-skills-design.md](../specs/2026-06-21-demo09-skills-design.md)

"""Demo #9 — knowledge-pack corpus generator + reference functions + scorer.

SINGLE SOURCE OF TRUTH: each core pack holds a Python `facts` table that BOTH
(a) renders the markdown reference tree on disk AND (b) drives the reference
function that computes the ground-truth answer. Markdown and answer key cannot
drift. Distractor packs are templated bulk to inflate the library to M packs.
"""
from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path
from typing import Any, Callable

import pandas as pd

# ---------------------------------------------------------------------------
# Data model
# ---------------------------------------------------------------------------

@dataclass
class Leaf:
    """A reference file in a pack's tree. May declare nested children."""
    name: str
    description: str                      # frontmatter description (catalog-visible)
    body: str                            # markdown body
    children: dict[str, "Leaf"] = field(default_factory=dict)


@dataclass
class CorePack:
    name: str                            # == directory name == frontmatter name
    description: str                     # SKILL.md catalog description (when to / not to use)
    overview: str                        # SKILL.md body
    references: dict[str, Leaf]          # top-level reference files
    reference_fn: Callable[[pd.DataFrame, dict], float]


# ---------------------------------------------------------------------------
# Pack 1: tax-by-region  (rule: net revenue is reported EX-VAT; rate is regional)
# ---------------------------------------------------------------------------
# Non-default rule: without the pack a model reports GROSS revenue (includes VAT)
# or guesses a flat rate -> wrong number. With the pack it divides by (1+rate)
# using the country's specific rate from references/<region>.md.

TAX_RATES = {  # SINGLE SOURCE OF TRUTH for both render + reference_fn
    # EU
    "DE": 0.19, "FR": 0.20, "ES": 0.21, "IT": 0.22, "NL": 0.21,
    # LATAM
    "BR": 0.17, "MX": 0.16, "AR": 0.21, "CO": 0.19, "CL": 0.19,
    # APAC
    "JP": 0.10, "AU": 0.10, "IN": 0.18, "SG": 0.09, "KR": 0.10,
}
_TAX_REGIONS = {
    "eu": ["DE", "FR", "ES", "IT", "NL"],
    "latam": ["BR", "MX", "AR", "CO", "CL"],
    "apac": ["JP", "AU", "IN", "SG", "KR"],
}


def _render_rate_table(codes: list[str]) -> str:
    rows = "\n".join(f"| {c} | {int(TAX_RATES[c]*100)}% |" for c in codes)
    return f"| country | vat_rate |\n|---|---|\n{rows}\n"


def _tax_reference_fn(df: pd.DataFrame, params: dict) -> float:
    country = params["country"]
    sub = df[(df["country"] == country) & (df["status"] == params.get("status", "shipped"))]
    gross = float((sub["quantity"].astype(float) * sub["unit_price_usd"].astype(float)).sum())
    rate = TAX_RATES[country]
    return round(gross / (1.0 + rate), 2)


_TAX_PACK = CorePack(
    name="tax-by-region",
    description=(
        "Use when a question asks for net revenue EX-VAT / ex-tax, or applies a "
        "country VAT/GST rate. NOT for gross revenue or for non-tax questions."
    ),
    overview=(
        "# Tax by region\n\n"
        "Reported **net revenue is always EX-VAT**: divide gross (quantity x "
        "unit_price_usd) by (1 + vat_rate). Rates are country-specific. For rate "
        "tables load reference `rates` (then the regional leaf `rates/eu`, "
        "`rates/latam`, or `rates/apac`). For special cases load `edge-cases`.\n"
    ),
    references={
        "rates": Leaf(
            name="rates",
            description="Index of VAT/GST rate tables by region. Children: eu, latam, apac.",
            body=(
                "# VAT/GST rates\n\nRates are grouped by region. Load the regional "
                "leaf for the country you need: `rates/eu`, `rates/latam`, `rates/apac`.\n"
                "Rounding: round net revenue to 2 decimals after dividing by (1+rate).\n"
            ),
            children={
                "eu": Leaf("eu", "VAT rates for EU countries (DE, FR, ES, IT, NL).",
                           "# EU VAT rates\n\n" + _render_rate_table(_TAX_REGIONS["eu"])),
                "latam": Leaf("latam", "VAT/IVA rates for LATAM (BR, MX, AR, CO, CL).",
                              "# LATAM rates\n\n" + _render_rate_table(_TAX_REGIONS["latam"])),
                "apac": Leaf("apac", "GST/consumption-tax rates for APAC (JP, AU, IN, SG, KR).",
                             "# APAC rates\n\n" + _render_rate_table(_TAX_REGIONS["apac"])),
            },
        ),
        "edge-cases": Leaf(
            name="edge-cases",
            description="B2B reverse-charge and digital-goods exceptions. Children: b2b, digital-goods.",
            body="# Tax edge cases\n\nSee `edge-cases/b2b` and `edge-cases/digital-goods`.\n",
            children={
                "b2b": Leaf("b2b", "Reverse-charge rule for B2B intra-EU sales.",
                            "# B2B reverse charge\n\nIntra-EU B2B sales are zero-rated; the buyer self-accounts.\n"),
                "digital-goods": Leaf("digital-goods", "Digital-goods VAT is charged at the buyer's country rate.",
                                      "# Digital goods\n\nDigital goods use the destination-country rate.\n"),
            },
        ),
    },
    reference_fn=_tax_reference_fn,
)

CORE_PACKS: dict[str, CorePack] = {
    "tax-by-region": _TAX_PACK,
}


# ---------------------------------------------------------------------------
# Rendering: pack object -> {relpath: markdown content} with frontmatter
# ---------------------------------------------------------------------------

def _frontmatter(name: str, description: str, child_refs: list[Leaf]) -> str:
    lines = ["---", f"name: {name}", f"description: {description}"]
    if child_refs:
        lines.append("references:")
        for c in child_refs:
            lines.append(f"  - name: {c.name}")
            lines.append(f"    description: {c.description}")
    lines.append("---\n")
    return "\n".join(lines)


def _render_leaf(rel_parent: str, key: str, leaf: Leaf, out: dict[str, str]) -> None:
    rel = f"{rel_parent}/{key}.md" if rel_parent else f"references/{key}.md"
    children = list(leaf.children.values())
    out[rel] = _frontmatter(leaf.name, leaf.description, children) + leaf.body
    base = rel[: -len(".md")]  # nested children live under references/<key>/<child>.md
    for ck, cl in leaf.children.items():
        _render_leaf(base, ck, cl, out)


def render_pack(pack: CorePack) -> dict[str, str]:
    """Return {relpath: content} for the whole pack tree (SKILL.md + references/*)."""
    out: dict[str, str] = {}
    top = list(pack.references.values())
    out["SKILL.md"] = _frontmatter(pack.name, pack.description, top) + pack.overview
    for key, leaf in pack.references.items():
        _render_leaf("", key, leaf, out)
    return out

Each pack follows the Task-1 template: a `facts` table (source of truth), a `_render_*` helper, a `reference_fn(df, params)`, a `CorePack(...)` with a 2-level reference tree, registered in `CORE_PACKS`. All rules are **plausible-but-non-default** (skipping the pack yields a different reasonable number).

The five packs and their rules:

1. **`returns-and-refunds`** — *rule:* an order with `status == 'returned'` counts as a refund and is **excluded** from net revenue; net = sum over `status == 'shipped'` only. (Default error: counting all non-pending, or including returned.) Facts: refund-window days table per channel (`web:30, retail:14, wholesale:60`) rendered in `references/windows/<channel>.md`; the reference_fn just filters to shipped, but the *question* asks "net revenue excluding refunds for channel X" so the channel-window leaf is the navigated fact establishing which orders are still within window — model **with** the pack learns "returned == refund == exclude".
   `reference_fn`: `round(sum(qty*price) where status=='shipped' and channel==params['channel'], 2)`.
2. **`revenue-recognition`** — *rule:* revenue is recognized **net of a platform fee** of a category-specific percentage (`electronics:0.08, apparel:0.12, home:0.10, grocery:0.04`). net = gross * (1 - fee). Facts: fee table per `product_category` in `references/fees/<category>.md`. `reference_fn`: `round(sum(qty*price)*(1-fee) for category==params['category'] and status=='shipped', 2)`.
3. **`discount-and-promo`** — *rule:* the stored `discount_pct` is a **display** discount; the **billable** discount is capped at a tier max per channel (`web:0.30, retail:0.20, wholesale:0.50`). net = sum(qty*price*(1 - min(discount_pct, cap))). Facts: cap table in `references/caps/<channel>.md`. `reference_fn` applies the cap.
4. **`payment-method-fees`** — *rule:* subtract a processor fee per `payment_method` (`card:0.029, paypal:0.034, bank_transfer:0.008, crypto:0.015`). net = sum(qty*price*(1-fee)). Facts: fee table in `references/processors/<method>.md`. `reference_fn` for `params['payment_method']` over shipped.
5. **`shipping-cost-allocation`** — *rule:* net contribution = revenue **minus** `shipping_usd`, but only shipping for `status=='shipped'` is allocated; free-shipping threshold per country (`order revenue >= threshold -> shipping treated as 0 cost to customer but still a cost to us`). `reference_fn`: `round(sum(qty*price - shipping_usd) for country==params['country'] and status=='shipped', 2)`. Facts: threshold table in `references/thresholds/<region>.md`.

def _returns_reference_fn(df, params):
    sub = df[(df["status"] == "shipped") & (df["channel"] == params["channel"])]
    return round(float((sub["quantity"].astype(float) * sub["unit_price_usd"].astype(float)).sum()), 2)

PLATFORM_FEES = {"electronics": 0.08, "apparel": 0.12, "home": 0.10, "grocery": 0.04}
def _revrec_reference_fn(df, params):
    fee = PLATFORM_FEES[params["category"]]
    sub = df[(df["status"] == "shipped") & (df["product_category"] == params["category"])]
    gross = float((sub["quantity"].astype(float) * sub["unit_price_usd"].astype(float)).sum())
    return round(gross * (1 - fee), 2)

DISCOUNT_CAPS = {"web": 0.30, "retail": 0.20, "wholesale": 0.50}
def _discount_reference_fn(df, params):
    cap = DISCOUNT_CAPS[params["channel"]]
    sub = df[(df["status"] == "shipped") & (df["channel"] == params["channel"])].copy()
    eff = sub["discount_pct"].astype(float).clip(upper=cap)
    net = sub["quantity"].astype(float) * sub["unit_price_usd"].astype(float) * (1 - eff)
    return round(float(net.sum()), 2)

PROCESSOR_FEES = {"card": 0.029, "paypal": 0.034, "bank_transfer": 0.008, "crypto": 0.015}
def _payment_reference_fn(df, params):
    fee = PROCESSOR_FEES[params["payment_method"]]
    sub = df[(df["status"] == "shipped") & (df["payment_method"] == params["payment_method"])]
    gross = float((sub["quantity"].astype(float) * sub["unit_price_usd"].astype(float)).sum())
    return round(gross * (1 - fee), 2)

def _shipping_reference_fn(df, params):
    sub = df[(df["status"] == "shipped") & (df["country"] == params["country"])]
    rev = sub["quantity"].astype(float) * sub["unit_price_usd"].astype(float)
    return round(float((rev - sub["shipping_usd"].astype(float)).sum()), 2)

@dataclass
class Question:
    id: str
    pack: str
    text: str                 # natural language; never names the pack mechanic explicitly
    params: dict
    leaf_path: str            # e.g. "rates/eu" — where the needed fact lives

def leaf_path_exists(pack_name: str, leaf_path: str) -> bool:
    node = CORE_PACKS[pack_name].references
    parts = leaf_path.split("/")
    cur = node.get(parts[0])
    for p in parts[1:]:
        if cur is None:
            return False
        cur = cur.children.get(p)
    return cur is not None

    "cohort-definitions", "channel-attribution", "fx-and-currency", "cogs-and-margin",
    "fiscal-calendar", "chargebacks-and-fraud", "inventory-valuation", "loyalty-points",
    "subscription-billing", "gift-cards", "marketplace-commissions", "warranty-claims",
    "price-matching", "bundle-pricing", "tax-exemptions", "credit-memos", "dunning",
    "deferred-revenue", "regional-rounding", "settlement-timing",
]  # extend/template to cover up to (50 - len(CORE_PACKS)) distractors

def _distractor_pack_files(name: str, rng_seed: str) -> dict[str, str]:
    """A realistic, bulky nested tree for a non-core domain. No reference_fn —
    no question targets it; its only jobs are token weight + retrieval confusion.
    Content is templated but domain-plausible and ~deterministic per name."""
    h = hashlib.sha256(rng_seed.encode()).hexdigest()
    regions = ["na", "emea", "apac", "latam"]
    files = {
        "SKILL.md": _frontmatter(
            name, f"Reference knowledge for {name.replace('-', ' ')}.",
            [Leaf(r, f"{name} parameters for region {r}.", "") for r in regions],
        ) + f"# {name}\n\nDomain rules and parameter tables for {name}. "
            f"Load the regional reference for specifics.\n",
    }
    for i, r in enumerate(regions):
        rows = "\n".join(
            f"| param_{j} | {int(h[(i+j) % len(h)], 16) * 7 % 100}% |" for j in range(40)
        )
        files[f"references/{r}.md"] = _frontmatter(
            r, f"{name} parameter table for {r}.", []
        ) + f"# {name} — {r}\n\n| parameter | value |\n|---|---|\n{rows}\n"
    return files

def _write_files(pack_dir: Path, files: dict[str, str]) -> None:
    for rel, content in files.items():
        fp = pack_dir / rel
        fp.parent.mkdir(parents=True, exist_ok=True)
        fp.write_text(content)

def materialize_corpus(out_dir: str, pack_count: int, seed: int) -> str:
    """Write `pack_count` packs to out_dir: all core packs always present, the
    rest filled with deterministic distractor packs. Returns out_dir."""
    import random
    root = Path(out_dir)
    if root.exists():
        import shutil; shutil.rmtree(root)
    root.mkdir(parents=True)
    for name, pack in CORE_PACKS.items():
        _write_files(root / name, render_pack(pack))
    n_distract = max(0, pack_count - len(CORE_PACKS))
    rng = random.Random(f"{pack_count}-{seed}")
    pool = list(_DISTRACTOR_DOMAINS)
    rng.shuffle(pool)
    chosen = pool[:n_distract]
    # if pool too small, suffix-number to reach n_distract deterministically
    while len(chosen) < n_distract:
        chosen.append(f"{pool[len(chosen) % len(pool)]}-{len(chosen)}")
    for name in chosen:
        _write_files(root / name, _distractor_pack_files(name, f"{name}-{seed}"))
    return out_dir

def corpus_token_estimate(corpus_dir: str) -> int:
    """~4 chars/token estimate over every .md file in the corpus."""
    total = sum(len(p.read_text()) for p in Path(corpus_dir).rglob("*.md"))
    return total // 4
```

import re

def _parse_number(text: str):
    if text is None:
        return None
    m = re.search(r"-?\d[\d,]*\.?\d*", str(text).replace(",", ""))
    return float(m.group()) if m else None

def score_skill_answer(question: "Question", produced: str, df: pd.DataFrame) -> dict:
    """correct: True/False, or None when the answer is empty/unparseable
    (NOT measured — never silently 0, per the Demo #8 honesty fix)."""
    want = CORE_PACKS[question.pack].reference_fn(df, question.params)
    got = _parse_number(produced)
    if got is None:
        return {"correct": None, "want": want, "got": None}
    ok = abs(got - want) <= 0.02 * max(1.0, abs(want))
    return {"correct": bool(ok), "want": want, "got": got}

"""Demo #9 — shared RAG helpers (framework-agnostic): chunk a corpus dir and
detect whether retrieval surfaced the leaf a question needs."""
from __future__ import annotations
from pathlib import Path

def chunk_corpus(corpus_dir: str) -> list[dict]:
    """One chunk per .md file (files are already leaf-sized < 64KB). Each chunk is
    tagged with its pack (top dir) and relpath so we can score retrieval hits."""
    root = Path(corpus_dir); out = []
    for md in sorted(root.rglob("*.md")):
        rel = md.relative_to(root)
        out.append({"pack": rel.parts[0], "relpath": str(Path(*rel.parts[1:])),
                    "text": md.read_text()})
    return out

def correct_chunk_hit(question, retrieved: list[dict]) -> bool:
    """True iff a retrieved chunk is the question's pack AND its needed leaf
    (references/<leaf_path>.md)."""
    want_rel = f"references/{question.leaf_path}.md"
    return any(r.get("pack") == question.pack and r.get("relpath") == want_rel
               for r in retrieved)
```
